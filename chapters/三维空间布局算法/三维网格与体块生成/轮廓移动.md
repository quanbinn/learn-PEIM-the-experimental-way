def move(data, p1, p2):
    length = len(data)

    xdiff = p2[0]-p1[0]
    ydiff = p2[1]-p1[1]   
    zdiff = p2[2]-p1[2]   

    for i in range(0,length):
        data[i][0] = data[i][0] + xdiff    
        data[i][1] = data[i][1] + ydiff   
        data[i][2] = data[i][2] + zdiff   
        
    return data

data = [[50,30,0], [285,77,0], [246,273,0], [10,226,0]]
p1 = [50,30,0]
p2 = [0,0,0]

new_data = move(data, p1, p2)
print(new_data)
