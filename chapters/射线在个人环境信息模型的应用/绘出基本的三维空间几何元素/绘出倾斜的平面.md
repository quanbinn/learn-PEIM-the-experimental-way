### 倾斜平面1

```python
import numpy as np
from matplotlib import pyplot as plt

fig = plt.figure()
ax = fig.gca(projection='3d')

x = np.linspace(0, 9, 10)
y = np.linspace(0, 9, 10)
X, Y = np.meshgrid(x, y)  
# print(X)
# print(Y)

# 平面 z=3 的部分
ax.plot_surface(X,
                Y,
                Z=X*0+3,
               )
  
# 显示图形
plt.show()
```

### 同时渲染出倾斜平面1, 2和3

```python
import numpy as np
from matplotlib import pyplot as plt

fig = plt.figure()
ax = fig.gca(projection='3d')

x = np.linspace(0, 9, 10)
y = np.linspace(0, 9, 10)
X, Y = np.meshgrid(x, y)  
# print(X)
# print(Y)

# 平面 z=3 的部分
ax.plot_surface(X,
                Y,
                Z=X*0+3,
               )

# 平面 z=2y 的部分
ax.plot_surface(X,
                Y=Y,
                Z=Y*2,
               )

# 平面 z=-2y + 10 部分
ax.plot_surface(X=X,
                Y=Y,
                Z=-Y*2+10,
               )
     
# 显示图形
plt.show()
```
